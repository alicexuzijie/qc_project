CREATE TABLE `T_SENSOR_QC_CONFIG_CAT` (
    `ID` BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '配置项编号',
    `CONFIG_CATEGORY` VARCHAR(100) NOT NULL COMMENT '配置项类别',
    PRIMARY KEY (`ID`),
    UNIQUE KEY `CONFIG_CATEGORY` (`CONFIG_CATEGORY`)
)  ENGINE=INNODB DEFAULT CHARSET=UTF8 COMMENT='配置项类别表';

CREATE TABLE `T_SENSOR_QC_CONFIG_GLOBAL_ITEM`(
        `ID` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '全局配置项编号',
        `CATEGORY_ID` bigint(20) NOT NULL COMMENT '类别编号',
        `CONFIG_ITEM` VARCHAR(100) NOT NULL COMMENT '全局配置项名称',
        `PARSE_TYPE` VARCHAR(100) NOT NULL COMMENT '全局配置项数据类别',
        `DESC` VARCHAR(100) NOT NULL COMMENT '全局配置项描述',
        PRIMARY KEY (`ID`),
        UNIQUE KEY `CONFIG_ITEM_GLOBAL` (`CONFIG_ITEM`),
        FOREIGN KEY(CATEGORY_ID) REFERENCES T_SENSOR_CONFIG_CAT(ID)
)       ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='全局配置项表';
